\begin{document}
    \centerline { 01.2 - Division Review}
    \centerline { Remember that $x \div y = \fracxy$, and that $x \div \frac1y = x \times y$}
    \begin{paracol}{2}
        \begin{enumerate}
            \setlength\itemsep{5em}
            \item $12 \div 4 = 3$
            \item $81 \div 9 = 9$
            \item $33 \div 11 = 3$
            \item $10 \div 2 = 5$
            \item $14 \div 2 = 7$
            \item $0 \div 4 = 0$
            \item$45 \div 3 = 15$
        \end{enumerate}
        \switchcolumn
        \begin{enumerate}
            \setlength\itemsep{5em}
            \setcounter{enumi}{7}
            \item $4 \div \frac13 = 4 \times 3 = 12$
            \item $10 \div \frac12 = 10 \times 2 = 20$
            \item $3 \div \frac15 = 3 \times 5 = 15$
            \item $ -10 \div -2 = 5$
            \item $-15 \div \frac15 = -15 \times 5 = 75$
            \item $\frac32 \div 4 = \frac32 \times \frac14$
            \item $\frac10 \div 3 = \frac{10}{3}$
        \end{enumerate}
    \end{paracol}
